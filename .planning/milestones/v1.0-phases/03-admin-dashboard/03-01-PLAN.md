       import type { NextRequest } from 'next/server';
       import { getIronSession } from 'iron-session';
       import { cookies } from 'next/headers';
       import { getSessionOptions, type SessionData } from '@/lib/session';

       export async function middleware(request: NextRequest) {
         const session = await getIronSession<SessionData>(await cookies(), getSessionOptions());
         if (!session.isLoggedIn) {

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser → /login (loginAction) | Untrusted password input crosses into server; compared against env secret |
| browser → /dashboard/* | Untrusted request must present a valid signed session cookie |
| server → client bundle | DASHBOARD_PASSWORD / DASHBOARD_SESSION_SECRET must never cross |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Spoofing | session cookie theft via XSS | mitigate | iron-session sets `httpOnly: true` in getSessionOptions — JS cannot read the cookie |
| T-03-02 | Elevation of Privilege | direct request to /dashboard/* without auth | mitigate | middleware.ts matcher `/dashboard/:path*` redirects to /login when `!session.isLoggedIn` |
| T-03-03 | Info Disclosure | shared password / session secret leaking to client | mitigate | Password read only in server action via `process.env`; `import 'server-only'` on auth.ts; session.ts reads `process.env` not exposed to client; NO default on either env var (fails loudly if misconfigured) |
| T-03-04 | Tampering | session cookie forged/modified | mitigate | iron-session encrypts (AES-256-CBC) + MACs the cookie with DASHBOARD_SESSION_SECRET (min 32 chars enforced by zod) |
| T-03-05 | Spoofing | password brute-force on /login | accept | Shared password is a v1 decision at small scale (single staff team); rate-limiting deferred. Min-8 length enforced. Documented as accepted risk. |
| T-03-06 | Info Disclosure | weak/empty session secret | mitigate | zod `.min(32)` on DASHBOARD_SESSION_SECRET with no default; getSessionOptions throws if absent |
</threat_model>
